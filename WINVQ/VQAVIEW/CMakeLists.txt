file_glob_enabled_languages_source_files()

add_executable(VQAVIEW)
target_include_directories(VQAVIEW
  PRIVATE
    ../INCLUDE
    ../INCLUDE/VQA32
    ../../WIN32LIB/INCLUDE
)
target_sources(VQAVIEW
  PRIVATE ${PROJECT_ANY_LANG_SOURCE_FILES}
)
target_sources(VQAVIEW
  PRIVATE
    FILE_SET HEADERS
    FILES ${PROJECT_ANY_LANG_IGNORES}
)
list(TRANSFORM CMAKE_CXX_IGNORE_GLOB_EXPRESSIONS
  PREPEND "../INCLUDE/"
  OUTPUT_VARIABLE PROJECT_CXX_IGNORE_GLOB_EXPRESSIONS
)
file(GLOB_RECURSE PROJECT_CXX_IGNORES
  ${PROJECT_CXX_IGNORE_GLOB_EXPRESSIONS}
)
target_sources(VQAVIEW
  PRIVATE
    FILE_SET include
    TYPE HEADERS
    BASE_DIRS "../INCLUDE"
    FILES ${PROJECT_CXX_IGNORES}
)
list(TRANSFORM CMAKE_CXX_IGNORE_GLOB_EXPRESSIONS
  PREPEND "../../WIN32LIB/INCLUDE/"
  OUTPUT_VARIABLE PROJECT_CXX_IGNORE_GLOB_EXPRESSIONS
)
file(GLOB_RECURSE PROJECT_CXX_IGNORES
  ${PROJECT_CXX_IGNORE_GLOB_EXPRESSIONS}
)
target_sources(VQAVIEW
  PRIVATE
    FILE_SET win32lib
    TYPE HEADERS
    BASE_DIRS "../../WIN32LIB/INCLUDE"
    FILES ${PROJECT_CXX_IGNORES}
)
install(TARGETS VQAVIEW)
